<h1>RESULTS</h1>

<h3>ArrayList</h3>

get()          = O(1);
set()          = O(1);
add()          = O(N);
add(,)         = O(N);
remove(i)      = O(N);
remove(obj)    = O(N);
add(0,)        = O(N);
add(size-1,0)  = O(N);

<h3>LinkedList</h3>

get()          = O(N);
set()          = O(N);
add()          = O(1);
add(,)         = O(N);
remove(i)      = O(N);
remove(obj)    = O(N);
add(0,)        = O(1);
add(size-1,0)  = O(1);
